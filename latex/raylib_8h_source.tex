\DoxyCodeLine{15 \textcolor{comment}{*       -\/ Flexible Materials system, supporting classic maps and PBR maps}}
\DoxyCodeLine{16 \textcolor{comment}{*       -\/ Animated 3D models supported (skeletal bones animation) (IQM, glTF)}}
\DoxyCodeLine{17 \textcolor{comment}{*       -\/ Shaders support, including Model shaders and Postprocessing shaders}}
\DoxyCodeLine{18 \textcolor{comment}{*       -\/ Powerful math module for Vector, Matrix and Quaternion operations: [raymath]}}
\DoxyCodeLine{19 \textcolor{comment}{*       -\/ Audio loading and playing with streaming support (WAV, OGG, MP3, FLAC, XM, MOD)}}
\DoxyCodeLine{20 \textcolor{comment}{*       -\/ VR stereo rendering with configurable HMD device parameters}}
\DoxyCodeLine{21 \textcolor{comment}{*       -\/ Bindings to multiple programming languages available!}}
